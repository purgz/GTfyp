\noindent $\Delta H$


i - rock, j - paper, k - scissors, N - i - j - k + players



H = -x(1-x)

$\Delta H = \Delta H(t+1) - \Delta H (t)$, \\
$\Delta H = -x_{t+1}(1-x_{t+1}) - (-x_{t}(1-x_t))$ \\
$\Delta H = -x_{t+1}(1-x_{t+1}) + x_{t}(1-x_t)$ \\
$\Delta H = x_{t}(1-x_t) -x_{t+1}(1-x_{t+1})$ \\
Rough equation: \\
$\langle \Delta H \rangle = \sum_{i,j,k}\Big(\Delta H_s - \Delta H_{s'}\Big)T^{s \rightarrow s'}$


\begin{equation}
\begin{aligned}
  \langle \Delta H \rangle &= \frac{1}{N^6} \sum_{i=1}^{N}\sum_{j=1}^{N}\sum_{k=1}^{N}\Big[(N-i-j-k)(1-N+i+j+k)(T^{R+} + T^{P+} + T^{S+}+ T^{+R}+ T^{+P} +T^{+S}) \\
  &- (N-i-j-k + 1)(-N+i+j+k)T^{R+} \\
  &- (N-i-j-k + 1)(-N + i + j + k)T^{P+} \\
  &- (N-i-j-k - 1)(2 - N + i + j + k)T^{+R} \\
  &- (N - i - j - k - 1)(2 - N + i + j + k)T^{+P} \\
  &- (N - i - j - k - 1)(2 - N + i + j + k)T^{+S}\Big]
\end{aligned}
\end{equation}



$q = 1 - x - y - z$, $p = N - i - j - k$


\begin{equation}
\begin{aligned}
  \langle \Delta H \rangle &= \frac{1}{N^6} \sum_{i=1}^{N}\sum_{j=1}^{N}\sum_{k=1}^{N}\Big[p(1-p)(T^{R+} + T^{P+} + T^{S+}+ T^{+R}+ T^{+P} +T^{+S}) \\
  &- (p + 1)(-p)(T^{R+} + T^{P+} + T^{S+}) \\
  &- (p-1)(2 -p)(T^{+R} + T^{+P} + T^{+S})\Big]
\end{aligned}
\end{equation}

\begin{aligned}
\langle \Delta H \rangle &= \int_{0}^{1}dx \int_{0}^{1} dy \int_{0}^{1}dz \Big[q(1-q)(T^{R+} + T^{P+} + T^{S+}+ T^{+R}+ T^{+P} +T^{+S}) \\
&- (q + \frac{1}{N})(1-q - \frac{1}{N})(T^{R+} + T^{P+} + T^{S+}) - (q - \frac{1}{N})(1-q + \frac{1}{N})(T^{+R} + T^{+P} + T^{+S}) \Big]
\end{aligned}
\end{equation}
