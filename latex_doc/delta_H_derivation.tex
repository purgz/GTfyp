\subsubsection{$\langle \Delta H_{SD} \rangle$ Derivation}

\noindent The equation $H = -q(1-q)$ defines a constant of motion for the SD part of the game,
where q is the fraction of players in the 4th strategy. Using the transition probabilities of 
the different process we can derive an expression for the expected change in H within the simplex.

\hfill

\noindent Where $i, j, k, N-i-j-k$ are the players playing R, P, S, and the 4th strategy respectively.



\hfill

\noindent $\Delta H = H(t+1) - H(t)$, \\
$\Delta H = -x_{t+1}(1-x_{t+1}) - (-x_{t}(1-x_t))$ \\
$\Delta H = -x_{t+1}(1-x_{t+1}) + x_{t}(1-x_t)$ \\
$\Delta H = x_{t}(1-x_t) -x_{t+1}(1-x_{t+1})$ \\

\hfill


$\langle \Delta H \rangle = \sum_{i,j,k}\Big(H_s - H_{s'}\Big)T^{s \rightarrow s'}$, $s$ is a 
particular state in the simplex.



\begin{equation} \label{delta_h_snowdrift}
\begin{aligned}
  \langle \Delta H_{SD} \rangle &= \frac{6}{N^5} \sum_{i=1}^{N}\sum_{j=1}^{N-i}\sum_{k=1}^{N-i-j}\Big[(N-i-j-k)(1-N+i+j+k)(T^{R+} + T^{P+} + T^{S+}+ T^{+R}+ T^{+P} +T^{+S}) \\
  &- (N-i-j-k + 1)(-N+i+j+k)T^{R+} \\
  &- (N-i-j-k + 1)(-N + i + j + k)T^{P+} \\
  &- (N-i-j-k - 1)(2 - N + i + j + k)T^{+R} \\
  &- (N - i - j - k - 1)(2 - N + i + j + k)T^{+P} \\
  &- (N - i - j - k - 1)(2 - N + i + j + k)T^{+S}\Big]
\end{aligned}
\end{equation}
The terms with transitions within the RPS simplex can be ignored as $q$ would not change between these states, therefore the term $H_s - H_{s'} = 0$ 


\hfill

$p = N - i - j - k$, the number of players playing the 4th strategy.


\begin{equation} \nonumber
\begin{aligned}
  \langle \Delta H_{SD} \rangle &= \frac{6}{N^5} \sum_{i=1}^{N}\sum_{j=1}^{N-i}\sum_{k=1}^{N-i-j}\Big[p(1-p)(T^{R+} + T^{P+} + T^{S+}+ T^{+R}+ T^{+P} +T^{+S}) \\
  &- (p + 1)(-p)(T^{R+} + T^{P+} + T^{S+}) \\
  &- (p-1)(2 -p)(T^{+R} + T^{+P} + T^{+S})\Big]
\end{aligned}
\end{equation}


\begin{equation} \nonumber
\begin{aligned}
\langle \Delta H_{SD} \rangle &= \frac{6}{N^5} \sum_{i=1}^{N}\sum_{j=1}^{N-i}\sum_{k=1}^{N-i-j} 
\Big[T^{R+}\left(p(1-p) + p(p+1)\right) + T^{+R}\left(p(1-p) - (p-1)(2-p)\right) \\
&+ T^{P+}\left(p(1-p) + p(p+1)\right) + T^{+P}\left(p(1-p) - (p-1)(2-p)\right) \\
&+ T^{S+}\left(p(1-p) + p(p+1)\right) + T^{+S}\left(p(1-p) - (p-1)(2-p)\right) 
\Big] \\
&= \frac{6}{N^5} \sum_{i=1}^{N}\sum_{j=1}^{N-i}\sum_{k=1}^{N-i-j} \Big[
2p\left(T^{R+} + T^{P+} + T^{S+}\right) + (p - p^2 - (3p - p^2 -2))(T^{+R} + T^{+P} + T^{+S})
\Big] \\
&= \frac{6}{N^5} \sum_{i=1}^{N}\sum_{j=1}^{N-i}\sum_{k=1}^{N-i-j}\Big[
2p\left(T^{R+} + T^{P+} + T^{S+}\right) + (2 - 2p)(T^{+R} + T^{+P} + T^{+S})
\Big] 
\end{aligned}
\end{equation}


\begin{equation} \nonumber
\langle \Delta H_{SD} \rangle = \frac{12}{N^5} \sum_{i=1}^{N}\sum_{j=1}^{N-i}\sum_{k=1}^{N-i-j} \Big[
p\left(T^{R+} + T^{P+} + T^{S+}\right) + (1 - p)(T^{+R} + T^{+P} + T^{+S})
\Big]
\end{equation}

\begin{equation}
\begin{aligned}
\langle \Delta H_{SD} \rangle &= \frac{12}{N^5} \sum_{i=1}^{N}\sum_{j=1}^{N-i}\sum_{k=1}^{N-i-j}
p\Big[ (T^{R+} - T^{+R}) + (T^{P+} - T^{+P}) +(T^{S+} - T^{+S}) \Big]+ T^{+R} + T^{+P} + T^{+S}
\end{aligned}
\end{equation}

The continuous limit, where $x = i/N$, $y = j/N$, $z = k/N$, and $q = p/N$, $p = Nq$ and $q = 1 - x - y - z$ leads to:

\begin{equation} \nonumber
\begin{aligned}
\langle \Delta H_{SD} \rangle &= \frac{12}{N^2} \int_{0}^{1}dx \int_{0}^{1-x} dy \int_{0}^{1-x-y}dz 
\Big[Nq\left[(T^{R+} - T^{+R}) + (T^{P+} - T^{+P}) +(T^{S+} - T^{+S})\right] \\
&+ (T^{+R} + T^{+P} + T^{+S}) \Big]
\end{aligned}
\end{equation}

Finally,

\begin{equation}
\begin{aligned}
\langle \Delta H_{SD} \rangle &= \frac{12}{N} \int_{0}^{1}dx \int_{0}^{1-x} dy \int_{0}^{1-x-y}dz 
\Big[q\left[(T^{R+} - T^{+R}) + (T^{P+} - T^{+P}) +(T^{S+} - T^{+S})\right]\Big] \\
&+ \frac{12}{N^2} \int_{0}^{1}dx \int_{0}^{1-x} dy \int_{0}^{1-x-y}dz\left[T^{+R} + T^{+P} + T^{+S}\right]
\end{aligned}
\end{equation}

This can then be solved numerically and the critical population values
can be found where $\langle \Delta H_{SD} \rangle = 0$.

\subsubsection{Analytical solution for the Local update process}

Considering the local update process, an exact analytical solution can be found for 
$\langle \Delta H_{SD} \rangle$.

\begin{align} \label{SimplifiedRPS}
\begin{bmatrix}
    0 & c & 1 & \gamma\\
    1 & 0 & c & \gamma\\
    c & 1 & 0 & \gamma\\
    \beta & \beta & \beta & 0
\end{bmatrix}
\end{align}

To simplify these derivations, I have fixed the values of $a = 0, b=1$, allowing
still to adjust the dynamics of the RPS game via $c$ and the SD game via $\gamma, \beta$
as used in all the results.

Evaluating the terms of the type $T^{R+} - T^{+R}$ for the local update process:

\begin{align}
T^{R+} - T^{+R} &= \frac{i}{N}\frac{N-i-j-k}{N}\left(\frac{1}{2} + \frac{w}{2}\frac{\pi_+ - \pi_R}{\Delta \pi_{\text{max}}}\right) - \frac{i}{N}\frac{N-i-j-k}{N}\left(\frac{1}{2} + \frac{w}{2}\frac{\pi_R - \pi_+}{\Delta \pi_{\text{max}}}\right) \nonumber \\
&= \frac{1}{2}\frac{i}{N}\frac{N-i-j-k}{N}w\left(\frac{\pi_+ - \pi_R}{\Delta \pi_{\text{max}}} 
- \frac{\pi_R - \pi_+}{\Delta \pi_{\text{max}}}\right)\nonumber \\
 &= w\frac{i}{N}\frac{N-i-j-k}{N}\left(\frac{\pi_+ - \pi_R}{\Delta \pi_{\text{max}}}\right)\nonumber
\end{align}

substituting the population frequencies:

\begin{equation}
T^{R+} - T^{+R} = wx(1-x-y-z)\frac{\pi_+ - \pi_R}{\Delta \pi_{\text{max}}}
\end{equation}

the difference for the other two strategies can simply be derived by swapping the corresponding
payoffs in the above expression.

Now considering the second integral with terms $T^{+R} + T^{+P} + T^{+S}$:

\begin{align}
T^{+R} + T^{+P} + T^{+S} &= \frac{i}{N}\frac{N-i-j-k}{N}\left(\frac{1}{2} 
+ \frac{w}{2}\frac{\pi_R - \pi_+}{\Delta \pi_{\text{max}}}\right) \\
&+ \frac{j}{N}\frac{N-i-j-k}{N}\left(\frac{1}{2}
 + \frac{w}{2}\frac{\pi_P - \pi_+}{\Delta \pi_{\text{max}}}\right) \\
 &+ \frac{k}{N}\frac{N-i-j-k}{N}\left(\frac{1}{2}
 + \frac{w}{2}\frac{\pi_S - \pi_+}{\Delta \pi_{\text{max}}}\right)
\end{align}

Again replacing with the frequencies:
\begin{align}
T^{+R} + T^{+P} + T^{+S} &= (1-x-y-z) \Bigg[ x\left(\frac{1}{2} 
+ \frac{w}{2}\frac{\pi_R - \pi_+}{\Delta \pi_{\text{max}}}\right) \\
&+ y\left(\frac{1}{2}
 + \frac{w}{2}\frac{\pi_P - \pi_+}{\Delta \pi_{\text{max}}}\right) 
 + z\left(\frac{1}{2}
 + \frac{w}{2}\frac{\pi_S - \pi_+}{\Delta \pi_{\text{max}}}\right) \Bigg] \\
&= \frac{1}{2}(1-x-y-z)\Bigg[x + y + z + \frac{w}{\Delta \pi_{\text{max}}}
\Big[x(\pi_R - \pi_+) + y(\pi_P - \pi_+) + z(\pi_S - \pi_+)\Big]
\Bigg]
\end{align}

Substituting in the expanded expressions for payoff differences e.g. $\pi_R - \pi_+$:

\begin{align}
T^{+R} + T^{+P} + T^{+S} &= 
\frac{1}{2}(1-x-y-z)\Bigg[x + y + z + \frac{w}{\Delta \pi_{\text{max}}} \\
&\Big[x(z + cy + \gamma(1 - x - y z) - \beta(x+y+z)) \\
&+ y(x + cz + \gamma(1 - x - y z) - \beta(x+y+z)) \\
&+ z(y + cx + \gamma(1 - x - y z) - \beta(x+y+z))\Big]
\Bigg]
\end{align}


Finally substituting everything back into $\langle \Delta H_{SD} \rangle$:


\begin{align}
\langle \Delta H_{SD} \rangle &= \frac{12}{N} \int_{0}^{1}dx \int_{0}^{1-x}dy \int_{0}^{1-x-y}dz 
(1-x-y-z)\Bigg[ \\
&\frac{w}{\Delta \pi_{\text{max}}} x(1-x-y-z)\left(\beta(x+y+z) - \gamma(1-x-y-z) -z -cy\right)\\
&\frac{w}{\Delta \pi_{\text{max}}} y(1-x-y-z)\left(\beta(x+y+z) - \gamma(1-x-y-z) -x -cz\right)\\
&\frac{w}{\Delta \pi_{\text{max}}} z(1-x-y-z)\left(\beta(x+y+z) - \gamma(1-x-y-z) -y -cx\right)\Bigg] \\
&+ \frac{6}{N^2} \int_{0}^{1}dx \int_{0}^{1-x}dy \int_{0}^{1-x-y}dz(1-x-y-z)\Bigg[ \\ 
&x + y + z + \frac{w}{\Delta \pi_{\text{max}}} 
\Bigg(x(z + cy + \gamma(1 - x - y z) - \beta(x+y+z)) \\
&+ y(x + cz + \gamma(1 - x - y z) - \beta(x+y+z)) \\
&+ z(y + cx + \gamma(1 - x - y z) - \beta(x+y+z))\Bigg)
\Bigg]
\end{align}


Solving the integrals leads to the exact analytical expression for local update process
$\langle \Delta H_{SD} \rangle$:

\begin{equation}
\langle \Delta H_{SD} \rangle_{LU} = 
\frac{42 \Delta \pi_{\text{max}} + 2w \gamma (7 - 6N) + (-1+4\beta - c)(-7 + 4N)w}{280N^2\Delta \pi_{\text{max}}}
\end{equation}
which is nicely expressed by the actual values in the payoff matrix for the game, 
here I have fixed the values of $a=0, b=1$ so that the expected drift can be written as 
an expression of $c, \beta, \gamma, N$ and $w$. $\Delta \pi_{\text{max}}$
is the maximum payoff difference which will depend on the choice of parameters, 
however it is a constant and usually defined as $b - c$ or $1-c$ when $b$ is fixed to 1
in this case as mostly we consider $0 < \gamma, \beta < 1$ so the maximal difference is
between the winning and losing case of the rock paper scissors sub game.

Since the critical population size is defined as a change in sign in 
$\langle \Delta H_{SD} \rangle$, the critical population size, $N_c$ can also be derived by setting
the above equation to 0 and solvign for $N$.


This leads to:
\begin{align}
N_c &= \frac{-42 \Delta \pi_{\text{max}} + w(28\beta - 7c  - 14 \gamma - 7)}{w(16 \beta - 4c -12\gamma - 4)}\\
&= \frac{-42 \Delta \pi_{\text{max}}}{w(16 \beta - 4c -12\gamma - 4)}
+ \frac{28 \beta - 7c - 14\gamma - 7}{16 \beta - 4c -12\gamma - 4}
\end{align}

I have compared this value to the numerically solved solutions of the triple integral 
and the results match perfectly. For both the values of the integral
and also the critical population size which was previously calculated using brentq root 
finding method.
Due to increasing complexity in the Moran process, I have used numerical methods to 
compute the value of the integral and critical population size instead of a closed form 
solution.

observation variables but this was a very long process for SD case so 
may be worth solving with numerical method.
Definitely not for the Moran process.

\subsubsection{$\langle \Delta H_{RPS} \rangle$ Derivation}

$\langle \Delta H \rangle$ within the RPS plane $H = -xyz$.

\begin{equation} \nonumber
\begin{aligned}
  \langle \Delta H_{RPS} \rangle &= \frac{2}{N^6} \sum_{i=1}^{N}\sum_{j=1}^{N-i}\sum_{k=1}^{N-i-j}\Big[ijk(T^{RP} + T^{RS} + T^{R+} + T^{PR} + T^{PS} + T^{P+} \\
  &+ T^{SR} + T^{SP} + T^{S+} + T^{+R} + T^{+P} + T^{+S}) \\
  &- k(i-1)(j+1)T^{RP} - (i-1)j(k+1)T^{RS} - jk(i-1)T^{R+} \\
  &- k(i+1)(j-1)T^{PR} - i(j-1)(k+1)T^{PS} - ik(j-1)T^{P+} \\
  &- (i+1)j(k-1)T^{SR} - i(j+1)(k-1)T^{SP} - ij(k-1)T^{S+} \\
  &- jk(i+1)T^{+R} - ik(j+1)T^{+P} - ij(k+1)T^{+S}\Big]
\end{aligned}
\end{equation}




\begin{equation} 
\begin{aligned}
  \langle \Delta H_{RPS} \rangle &= \frac{2}{N^6} \sum_{i=1}^{N}\sum_{j=1}^{N-i}\sum_{k=1}^{N-i-j}
  \Big[k(j-i)(T^{RP} - T^{PR}) + j(k-i)(T^{RS} - T^{SR}) + i(j-k)(T^{SP} - T^{PS}) \\
  &+ k(T^{RP} + T^{PR}) + j(T^{RS} + T^{SR}) + i(T^{SP} + T^{PS}) \\
  &+ jk(T^{R+} - T^{+R}) + ik(T^{P+} - T^{+P}) + ij(T^{S+} - T^{+S})
  \Big]
\end{aligned}
\end{equation}
Compared to the derivation of simply the RPS case in\cite{PhysRevLett.100.058104_cyclic}, this has the additional
terms including the difference in tranition probabilities in and out of the 4th strategy.


\begin{equation} \nonumber
\begin{aligned}
  \langle \Delta H_{RPS} \rangle &= \frac{2}{N^3} \int_{0}^{1}dx \int_{0}^{1-x} dy \int_{0}^{1-x-y}dz 
  \Big[N^2z(y-x)(T^{RP} - T^{PR}) + N^2y(z-x)(T^{RS} - T^{SR}) \\
  &+ N^2x(y-z)(T^{SP} - T^{PS}) \\
  &+ Nz(T^{RP} + T^{PR}) + Ny(T^{RS} + T^{SR}) + Nx(T^{SP} + T^{PS}) \\
  &+ N^2xz(T^{P+} - T^{+P}) + N^2yz(T^{R+} - T^{+R}) + N^2xy(T^{S+} - T^{+S})
  \Big]
\end{aligned}
\end{equation}

\begin{equation}
\begin{aligned}
  \langle \Delta H_{RPS} \rangle &= \frac{2}{N} \int_{0}^{1}dx \int_{0}^{1-x} dy \int_{0}^{1-x-y}dz 
  \Big[z(y-x)(T^{RP} - T^{PR}) + y(z-x)(T^{RS} - T^{SR}) \\
  &+ x(y-z)(T^{SP} - T^{PS}) + xz(T^{P+} - T^{+P})+ yz(T^{R+} - T^{+R}) + xy(T^{S+} - T^{+S})
  \Big] \\
  &+ \frac{2}{N^2} \int_{0}^{1}dx \int_{0}^{1-x} dy \int_{0}^{1-x-y}dz\Big[
  z(T^{RP} + T^{PR}) + y(T^{RS} + T^{SR}) + x(T^{SP} + T^{PS})\Big]
\end{aligned}
\end{equation}


\subsubsection{Analytical solution for the Local update process $\langle \Delta H_{RPS} \rangle$}


An analytical solution for $\langle \Delta H_{RPS} \rangle$ can also be 
derived for for the local update process, this will require 
the simplified terms of the forms $T^{RP} - T^{PR}$ and
$T^{RP} + T^{PR}$ within the RPS game, and the terms of form 
$T^{R+} - T^{+R}$ have already been derived in the solution for 
$\langle \Delta H_{SD} \rangle$.


\begin{align}
T^{RP} - T^{PR} &= \left(\frac{1}{2} + \frac{w}{2}\left(\frac{\pi_P - \pi_R}{\Delta \pi_{max}}\right)\right)\frac{i}{N} \frac{j}{N} 
- \left(\frac{1}{2} + \frac{w}{2}\left(\frac{\pi_R - \pi_P}{\Delta \pi_{max}}\right)\right)\frac{i}{N} \frac{j}{N} \nonumber \\
&= \frac{i}{N} \frac{j}{N} \left[\frac{w}{2\Delta \pi_{max}} \left(2(\pi_P - \pi_R)\right)\right] \nonumber \\
&= \frac{w}{\Delta \pi_{max}}xy(\pi_P - \pi_R)
\end{align}
where population counts e.g. $i/N, j/N$ are replaced with frequencies $x, y$ respectively. 
Similarly the other 2 terms within the RPS game can be derived leading to:

\begin{align}
T^{RS} - T^{SR} = \frac{w}{\Delta \pi_{max}}xz(\pi_S - \pi_R)\nonumber \\
T^{SP} - T^{PS} = \frac{w}{\Delta \pi_{max}}zy(\pi_P - \pi_S)
\end{align}

Substituting in the expanded payoff differences from the payoff matrix:
\begin{align}
T^{RP} - T^{PR} = \frac{w}{\Delta \pi_{max}}xy(x - z + c(z-y))\nonumber \\
T^{RS} - T^{SR} = \frac{w}{\Delta \pi_{max}}xz(y - z + c(x-y))\nonumber \\
T^{SP} - T^{PS} = \frac{w}{\Delta \pi_{max}}zy(x - y + c(z-x)) \nonumber
\end{align}

The terms of form $T^{R+} - T^{+R}$ have already been derived in section for $\langle \Delta H_{SD} \rangle$.

Finally for the terms of order $1/N^2$:

\begin{align}
T^{RP} + T^{PR} &= \left(\frac{1}{2} + \frac{w}{2}\left(\frac{\pi_P - \pi_R}{\Delta \pi_{max}}\right)\right)\frac{i}{N} \frac{j}{N} 
+ \left(\frac{1}{2} + \frac{w}{2}\left(\frac{\pi_R - \pi_P}{\Delta \pi_{max}}\right)\right)\frac{i}{N} \frac{j}{N} \nonumber \\
&= \frac{w}{2\Delta \pi_{max}}xy\left(\pi_P - \pi_R + \pi_R - \pi_p \right) + xy \nonumber\\
&= xy \nonumber
\end{align}
The terms containing $w$ and payoffs for each strategy cancel out for these terms within RPS, leaving just:

\begin{align}
T^{RP} + T^{PR} &= xy \nonumber \\
T^{RS} + T^{SR} &= xz \nonumber \\
T^{RS} + T^{SR} &= zy \nonumber \\\nonumber
\end{align} 


\begin{align}
\langle \Delta H_{RPS} \rangle_{LU} &= \frac{2}{N} \int_{0}^{1}dx \int_{0}^{1-x} dy \int_{0}^{1-x-y}dz 
\Big[z(y-x)(\frac{w}{\Delta \pi_{max}}xy(x - z + c(z-y)))  \nonumber\\
&+ y(z-x)(\frac{w}{\Delta \pi_{max}}xz(y - z + c(x-y))) + x(y-z)(\frac{w}{\Delta \pi_{max}}zy(x - y + c(z-x)))  \nonumber\\
&+ xz(\frac{w}{\Delta \pi_{\text{max}}} y(1-x-y-z)\left(\beta(x+y+z) - \gamma(1-x-y-z) -x -cz\right))  \nonumber\\
&+ yz(\frac{w}{\Delta \pi_{\text{max}}} x(1-x-y-z)\left(\beta(x+y+z) - \gamma(1-x-y-z) -z -cy\right)) \nonumber \\
& + xy(\frac{w}{\Delta \pi_{\text{max}}} z(1-x-y-z)\left(\beta(x+y+z) - \gamma(1-x-y-z) -y -cx\right))
\Big] \nonumber \\
&+ \frac{2}{N^2} \int_{0}^{1}dx \int_{0}^{1-x} dy \int_{0}^{1-x-y}dz\left(
3xyz\right) \nonumber
\end{align}

The second integral equals:
\begin{align}
\frac{2}{N^2} \int_{0}^{1}dx \int_{0}^{1-x} dy \int_{0}^{1-x-y}dz\left(
3xyz\right) &= \frac{6}{N^2} \int_{0}^{1}dx \int_{0}^{1-x} dy \int_{0}^{1-x-y}dz\left(
xyz\right) \\
&= \frac{6}{N^2} \frac{1}{720} = \frac{1}{120N^2}
\end{align}


\begin{align}
\langle \Delta H_{RPS} \rangle_{LU} &= \frac{w}{\Delta \pi_{max}} \frac{2}{N} \int_{0}^{1}dx \int_{0}^{1-x} dy \int_{0}^{1-x-y}dz 
xyz \Big[(y-x)(x - z + c(z-y))  \nonumber\\
&+ (z-x)(y - z + c(x-y)) + (y-z)(x - y + c(z-x)) \nonumber\\
&+ (1-x-y-z)\left(\beta(x+y+z) - \gamma(1-x-y-z) -x -cz\right) \nonumber\\
&+ (1-x-y-z)\left(\beta(x+y+z) - \gamma(1-x-y-z) -z -cy\right) \nonumber \\
& + (1-x-y-z)\left(\beta(x+y+z) - \gamma(1-x-y-z) -y -cx\right)
\Big] \nonumber \\
&+ \frac{1}{120N^2} \nonumber \\
&= \frac{1}{120N^2} - \frac{w(2 + \gamma - 3 \beta  + 2c)}{3360 N \Delta \pi_{max}}
\end{align}

By setting this expression equal to zero, a critical population size for RPS can be found:

\begin{equation}
N_{RPS} = \frac{28 \Delta \pi_{max}}{w(2 + \gamma - 3 \beta + 2c)}
\end{equation}

\subsubsection{$\langle \Delta H_4 \rangle$ Derivation}



All 4 strategies, $H = -xyz(1-x-y-z)$, Therefore $\langle \Delta H_4 \rangle$ for the population counts can be defined as:


\begin{equation} \label{delta_h_4}
\begin{aligned}
  \langle \Delta H_{4} \rangle &= \frac{6}{N^7} \sum_{i=1}^{N} \sum_{j=1}^{N-i}  \sum_{k=1}^{N-i-j} \Big[ijk(N-i-j-k)
  (T^{RP} + T^{RS} + T^{R+} + T^{PR} + T^{PS} + T^{P+} \\
  &+ T^{SR} + T^{SP} + T^{S+} + T^{+R} + T^{+P} + T^{+S}) - (i-1)(j+1)k(N-i-j-k)T^{RP} \\
  &- (i-1)j(k+1)(N-i-j-k)T^{RS} - (i-1)jk(N-i-j-k + 1)T^{R+} \\
  &- (i+1)(j-1)k(N-i-j-k)T^{PR} - i(j-1)(k+1)(N-i-j-k)T^{PS} \\
  &- i(j-1)k(N-i-j-k + 1)T^{P+} - (i+1)j(k-1)(N-i-j-k)T^{SR} \\
  &- i(j+1)(k-1)(N-i-j-k)T^{SP} - ij(k-1)(N-i-j-k + 1)T^{S+} \\
  &- (i+1)jk(N - 1 -i-j-k)T^{+R} - i(j+1)k(N - 1-i-j-k)T^{+P} - ij(k+1)(N - 1-i-j-k)T^{+S}\Big]
\end{aligned}
\end{equation}



Normalization $\frac{6}{N^7}$, as its over the whole simplex (pyramid volume 1/6) and triple summation, ($N^3$), then 4 populations $N^4$.

\hfill

Let $p$ = N - i - j - k (4th strategy count):


\begin{equation} \label{delta_h_4}
\begin{aligned}
  \langle \Delta H_{4} \rangle &= \frac{6}{N^7} \sum_{i=1}^{N} \sum_{j=1}^{N-i}  \sum_{k=1}^{N-i-j} \Big[ijkp
  (T^{RP} + T^{RS} + T^{R+} + T^{PR} + T^{PS} + T^{P+} \\
  &+ T^{SR} + T^{SP} + T^{S+} + T^{+R} + T^{+P} + T^{+S}) - (i-1)(j+1)kpT^{RP} \\
  &- (i-1)j(k+1)pT^{RS} - (i-1)jk(p+ 1)T^{R+} \\
  &- (i+1)(j-1)kpT^{PR} - i(j-1)(k+1)pT^{PS} \\
  &- i(j-1)k(p + 1)T^{P+} - (i+1)j(k-1)pT^{SR} \\
  &- i(j+1)(k-1)pT^{SP} - ij(k-1)(p + 1)T^{S+} \\
  &- (i+1)jk(p - 1)T^{+R} - i(j+1)k(p - 1)T^{+P} - ij(k+1)(p - 1)T^{+S}\Big]
\end{aligned}
\end{equation}


Looking at one pair of transitions in and out of the same state. ($R
\rightarrow P, P \rightarrow R$) this can be rewritten as before for 
$\langle \Delta H_{RPS} \rangle$ as differences of transition probabilities
in and out of a state.


\begin{equation} \label{delta_h_4_pairs}
\begin{aligned}
  \langle \Delta H_{4} \rangle &= \frac{6}{N^7} \sum_{i=1}^{N} \sum_{j=1}^{N-i}  \sum_{k=1}^{N-i-j} \Big[
  kp(j-i)(T^{RP} - T^{PR}) + jp(k-i)(T^{RS} - T^{SR}) \\
  &+ ip(j-k)(T^{SP} - T^{PS}) + kp(T^{RP} + T^{PR}) + jp(T^{RS} + T^{SR}) \\
  &+ ip(T^{SP} + T^{PS}) + jk(p-i)(T^{R+} - T^{+R}) + ik(p-j)(T^{P+} - T^{+P}) \\
  &+ ij(p-k)(T^{S+} - T^{+S}) + jk(T^{R+} + T^{+R}) + ik(T^{P+} + T^{+P}) + ij(T^{S+} + T^{+S})\Big]
\end{aligned}
\end{equation}

As before, converting this to the integral form where $x= i / N, y = j/N, z=k/N$ and $q=p/N, p=Nq$ where $q=1-x-y-z$ leads to:

\begin{equation}
\begin{aligned}
  \langle \Delta H_{4} \rangle &= \frac{6}{N^4} \int_{0}^{1}dx \int_{0}^{1-x} dy \int_{0}^{1-x-y}dz 
  \Big[N^3zq(y - x)(T^{RP} - T^{PR}) + N^3yq(z - x)(T^{RS} - T^{SR}) \\
  &+ N^3xq(y-z)(T^{SP} - T^{PS}) + N^2zq(T^{RP} + T^{PR}) + N^2yq(T^{RS} + T^{SR}) + N^2xq(T^{SP} + T^{PS}) \\
  &+ N^3yz(q-x)(T^{R+} - T^{+R}) + N^3xz(q-y)(T^{P+} - T^{+P}) + N^3xy(q-z)(T^{S+} - T^{+S}) \\
  &+ N^2yz(T^{R+} + T^{+R}) + N^2xz(T^{P+} + T^{+P}) + N^2xy(T^{S+} + T^{+S})\Big]
\end{aligned}
\end{equation}


\begin{equation}\label{final_delta_H_4}
\begin{aligned}
  \langle \Delta H_{4} \rangle &= \frac{6}{N} \int_{0}^{1}dx \int_{0}^{1-x} dy \int_{0}^{1-x-y}dz 
  \Big[zq(y - x)(T^{RP} - T^{PR}) + yq(z - x)(T^{RS} - T^{SR}) +xq(y-z)(T^{SP} - T^{PS}) \\ 
  &+ yz(q-x)(T^{R+} - T^{+R}) + xz(q-y)(T^{P+} - T^{+P}) + xy(q-z)(T^{S+} - T^{+S})\Big] \\
  &+ \frac{6}{N^2} \int_{0}^{1}dx \int_{0}^{1-x} dy \int_{0}^{1-x-y}dz\Big[zq(T^{RP} + T^{PR}) + yq(T^{RS} + T^{SR}) + xq(T^{SP} + T^{PS}) \\
  &+ yz(T^{R+} + T^{+R}) + xz(T^{P+} + T^{+P}) + xy(T^{S+} + T^{+S})\Big]
\end{aligned}
\end{equation}

Final derivation of $\langle \Delta H_4 \rangle$. There are clear similarities to the other two expected values
$\langle \Delta H_{SD} \rangle, \langle \Delta_{RPS} \rangle$ with some terms missing due to the different
observation variable $H$ used but they can all be rewritten in terms of differences of transition  probabilities
into and out of state (e.g. ($T^{RP} - T^{PR}$)) in an integral scaled by $1/N$ plus positive terms in a second integral scaled by $1/N^2$.
