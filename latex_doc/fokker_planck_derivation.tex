\noindent Master equation for Rock
\begin{equation}
  \begin{aligned}
  P^{\tau + 1}(i) - P^\tau(i) &= P^\tau(i-1)T^{PR}(i-1) - P^\tau(i)T^{PR}(i) + P^\tau(i-1)T^{SR}(i-1) - P^\tau(i)T^{SR}(i) \\
  &+ P^\tau(i-1)T^{+R}(i-1) - P^\tau(i)T^{+R}(i) + P^\tau(i+1)T^{RP}(i+1) - P^\tau(i)T^{RP}(i) \\
  &+ P^\tau(i+1)T^{RS} - P^\tau(i)T^{RS}(i) + P^\tau(i+1)T^{R+} - P^\tau(i)T^{R+}(i)
  \end{aligned}
\end{equation}
Where $i$ is the number of rock players.

\begin{equation}
  T(x\pm\frac{1}{N}) = T(x) \pm \frac{1}{N}T'(x) + \frac{1}{2N^2}T''(x)\dots \nonumber
\end{equation}
\begin{equation}
  P(x\pm\frac{1}{N}) = P(x) \pm \frac{1}{N}P'(x) + \frac{1}{2N^2}P''(x)\dots \nonumber
\end{equation}


\begin{equation}
\begin{aligned}
P^{\tau + 1}(i) - P^\tau(i) 
&= (P-\frac{1}{N}P' + \frac{1}{2N^2}P'')(T^{PR} - \frac{1}{N}T^{PR'} + \frac{1}{2N^2}T^{PR''}) - PT^{PR} \\
&+ (P-\frac{1}{N}P' + \frac{1}{2N^2}P'')(T^{SR} - \frac{1}{N}T^{SR'} + \frac{1}{2N^2}T^{SR''}) - PT^{SR} \\
&+ (P-\frac{1}{N}P' + \frac{1}{2N^2}P'')(T^{+R} - \frac{1}{N}T^{+R'} + \frac{1}{2N^2}T^{+R''}) - PT^{+R} \\
&+ (P+\frac{1}{N}P' + \frac{1}{2N^2}P'')(T^{RP} + \frac{1}{N}T^{RP'} + \frac{1}{2N^2}T^{RP''}) - PT^{RP} \\
&+ (P+\frac{1}{N}P' + \frac{1}{2N^2}P'')(T^{RS} + \frac{1}{N}T^{RS'} + \frac{1}{2N^2}T^{RS''}) - PT^{RS} \\
&+ (P+\frac{1}{N}P' + \frac{1}{2N^2}P'')(T^{R+} + \frac{1}{N}T^{R+'} + \frac{1}{2N^2}T^{R+''}) - PT^{R+}
\end{aligned}
\end{equation}

\begin{equation}
\begin{aligned}
P^{\tau+1}(i)-P^\tau(i)
&= -\frac{1}{N}\Big[
      P\big(
          T^{PR'} + T^{SR'} + T^{+R'}
        - T^{RP'} - T^{RS'} - T^{R+'}
      \big)
\\
&\qquad\qquad
    + P'\big(
          T^{PR} + T^{SR} + T^{+R}
        - T^{RP} - T^{RS} - T^{R+}
      \big)
    \Big]
\\[6pt]
&\quad
 + \frac{1}{2N^2}\Big[
      P\big(T^{PR''}+T^{SR''}+T^{+R''} + T^{RP''} +T^{RS''}+T^{R+''} \big) \\
      &\qquad\qquad 
      + 2P'\big(T^{PR'}+T^{SR'}+T^{+R'}+T^{RP'} +T^{RS'}+T^{R+'} \big) \\
      &\qquad\qquad
      + P''\big(T^{PR}+T^{SR}+T^{+R}+T^{RP} +T^{RS}+T^{R+}\big)
    \Big]
\\
&\quad=\frac{1}{N}\Big[
    -\big(P\big(
          T^{PR} + T^{SR} + T^{+R}
        - T^{RP} - T^{RS} - T^{R+}
      \big)\big)'
\\[6pt]
&\quad
 + \frac{1}{2}\big(
      P\big(\frac{T^{PR}+T^{SR}+T^{+R} + T^{RP} +T^{RS}+T^{R+}}{N}\big)
    \big)'' \Big]
\end{aligned}
\end{equation}


\hfill

\par \noindent
From these we can form the fokker planck equations for each strategies population $(x \in \{i,j,k\})$:

\begin{equation}
  \frac{d}{dt}P^\tau(x) = \frac{1}{N}\Big[-\frac{d}{d x}\big(a(x)P^\tau(x) \big) + \frac{1}{2} \frac{d^2}{d x^2}\big(b^2(x)P^\tau(x) \big)\Big]
\end{equation}
Where $a(x)$ corresponds to the deterministic replicator dynamics
as previously defined e.g. the payoff comparison dynamics, and 
$b(x)$ is a noise correction term influenced by the 
transition probabilities and the population size $N$.

\hfill
\\
\noindent This results in the following Langevin equation
with noise term $b(x)\xi$ (stochastic differential equation):
\begin{equation}
  \dot{x} = a(x) + b(x)\xi
\end{equation}
$a(x)$ is the deterministic replicator equation as $N \rightarrow \infty$ noise dissapears ($b^2(x)$ is divided by N).


\subsubsection{Link to deterministic equations $\lim_{N\to\infty}$}



In the standard case where $N$ is very large or $N \rightarrow \infty$, 
the noise term dissapears as it has a factor of $\sqrt{\frac{1}{N}}$
which leaves just:
\begin{equation}
\dot{x} = a(x)
\end{equation}

For the Local update process, equation
collapses to the standard replicator dynamics traditionally
used in evolutionary game theory, with an extra constant 
scaling factor consisting of $w$, the strength of 
selection.


\begin{align}
\dot{x} &= T^{RS} + T^{PS} + T^{+S} - T^{SR} - T^{SP} - T^{S+} \nonumber\\
&= \frac{w}{{\Delta \pi_\text{max}}} x_i(\pi_i(x) - \langle \pi(x)\rangle)
\end{align}

Exact term for the difference between $\pi_i$ and average payoff
\begin{align}
2 a x y + 2 a x z + 2 a y z - a y - a z - b x y \\
- b x z - b y z + b z + \beta x^{2} + 2 \beta x y \\
+ 2 \beta x z - \beta x + \beta y^{2} + 2 \beta y z \\
- \beta y + \beta z^{2} - \beta z - c x y - c x z \\
- c y z + c y + \gamma x^{2} + 2 \gamma x y \\
+ 2 \gamma x z - 2 \gamma x + \gamma y^{2} \\
+ 2 \gamma y z - 2 \gamma y + \gamma z^{2} \\
- 2 \gamma z + \gamma
\end{align}



\par
For the Moran process, the infinite population dynamics reduce to a scaled version of the ordinary replicator equation by the factor $\frac{1}{\Gamma + \langle \pi(x) \rangle}$
\begin{equation}
 a(x) = \frac{1}{\Gamma + \langle \pi(x) \rangle} x_i(\pi_i(x) - \langle \pi(x) \rangle), \Gamma = \frac{1 - w}{w}
\end{equation}

\begin{equation} \nonumber
\begin{aligned}
\langle \pi(x) \rangle &= x \left(a x + b z + c y + \gamma \left(- x - y - z + 1\right)\right) \\
&+ y \left(a y + b x + c z + \gamma \left(- x - y - z + 1\right)\right) \\
&+ z \left(a z + b y + c x + \gamma \left(- x - y - z + 1\right)\right) \\
&+ \left(x \left(a + \beta\right) + y \left(a + \beta\right) + z \left(a + \beta\right)\right) \left(- x - y - z + 1\right)
\end{aligned}
\end{equation}
